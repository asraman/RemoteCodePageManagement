/*
*  Package Description:
* ======================
*  This package defines DgbShim class which can be used by .NET applications.
*  In this project, DgbShim is used in WPF project to allow making calls from
*  WPF C# code to Executive native C++ code. The DgbShim is a mangaged (.NET)
*  wrapper around IExecutive interface which can make calls to native C++
*  classes and, in the same time, can be called by any .NET code (C# here).
*
*  Public Interface:
* ===================
*  DgbShim obj
*  obj.start(args);
*  obj.initialization(argc, argv);
*  obj.vectorToList(files);
*
*  Required Files:
* =================
*  DgbShim.h DgbShim.cpp IExecutive.h
*
*  Build Command:
* ================
*  msbuild DgbShim.vcxproj
*
*  Maintainence History:
* =======================
*  ver 1.0 - April 5th 2019
*    - first release
*
*/

// 'public ref' identifier tells the compiler this is


// a managed class, not native C++ class. this class


// has to be managed for the C# code in WPF to be able


// to use it. You CAN define native classes here but in


// this demo there is no point to do so. 

public ref class DgbShim
{
public:
	DgbShim();
	~DgbShim();
	
//void start();

	std::vector<std::string> initialization(int argc, char ** argv);
	List<String^>^ start(List<String^>^ args);
	List<String^>^ vectorToList(const std::vector<std::string>& files);
private:
	
// convert System::String to std::string

	std::string sysStrToStdStr(String^ str);
	
// convert std::string to System::String

	String^ stdStrToSysStr(const std::string& str);
	
// this will point to CppClass when instantiated through the ObjectFactory

	IExecutiveClass* client;
};
